点击发起

ScopedModelDescendant<CounterModel>(
        builder: (context, child, model) {
          return FloatingActionButton(
            onPressed: () {
               model.increment(2);
            },
            tooltip: 'add',
            child: Icon(Icons.add),
          );
        },
      ),

自动接收model变化

ScopedModelDescendant<CounterModel>(
              builder: (context, child, model) {
                return Text(
                  '${model.counter.toString()} 次了',
                  style: TextStyle(
                    color: Colors.red,
                    fontSize: 33.0,
                  ),
                );
              },
            ),
